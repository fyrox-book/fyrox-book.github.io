with low effort. To start off we will use the simplest framework:

        Self { }
    }

    fn on_ui_message(&mut self, engine: &mut Engine, message: UiMessage) {

        // Here we'll handle messages from user interface.
        //
    }
}

fn main() {
    Framework::<Game>::new()
        .unwrap()
        .title("User Interface")
        .run();
}
